                For i = 0 To UBound(acentos)

                End Function
                Function NomePermitido(nome)
                Dim nomeNormalizado, i
                nomeNormalizado = UCase(RemoveAcentos(nome))
                For i = 0 To UBound(nomesPermitidos)
                If nomeNormalizado = nomesPermitidos(i) Then
                NomePermitido = True
                Exit Function
                End If
                Next
                NomePermitido = False
                End Function
                Dim nomeUsuario
                nomeUsuario = InputBox("Digite seu nome:")
                If NomePermitido(nomeUsuario) Then
                MsgBox "Entrada permitida!"
                Else
                MsgBox "Entrada não permitida!"
                End If
